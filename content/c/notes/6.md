```c
// We can use '<:, :>' in place of '[, ]' and '<%, %>'' in place of '{, }'
#include<stdio.h>
int main()
<%
    int arr <:10:>;
    arr<:0:> = 1;
    printf("%d", arr<:0:>);
  
    return 0;
%>

// Output: 1
```
